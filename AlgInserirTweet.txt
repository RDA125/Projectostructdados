algoritmo "Inserir um Tweet pelo utilizador"
objectivo: Inserir Tweet para um ficheiro com o nome do utilizador.

//função
funcao InserirTweet(nome:caractere)
var
	likes,p:inteiro
	f:file stream
	topico,tweet,linhas[100]:caracter

inicio
	
	likes <- 0
	
	f <- Abrir("tweets.txt","r")
	linhas <- f.lerLinhas()
	f.fechar()

	p <- tamanho(linhas)

	Por cada linha em Linhas fazer
	
		linha <- linha.remover("\n")
		linha <- linha.remover("\n")
		linha <- linha.separa("-")
	
		se (p == linha[0]) então
			p++
		fimse

	fimPorCada
	
	f <- Abrir("tweets.txt","a")

	Escreve("Insira tópico: ")
	ler(topico)
	
	fazer
		Escreve(Insira tweet)
		ler(tweet)

	enquanto(Tamanho(tweet)>280)

	FicheiroEscreve(f,"%d-%s-%s-%s-%d\n",p,nome,topico,tweet,likes)
	f.fechar()
	
	Escreve("Tweet inserido com sucesso")
	
fimfuncao

//algoritmo
inicio
	
	InserirTweet("João")
fim
